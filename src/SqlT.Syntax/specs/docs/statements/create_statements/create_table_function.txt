<create_table_function> ::=
	CREATE [ OR ALTER ] FUNCTION [ schema_name. ] function_name   
	( [ { @parameter_name [ AS ] [ type_schema_name. ] parameter_data_type   
		[ = default ] [ READONLY ] }   
		[ ,...n ]  
	  ]  
	)  
	RETURNS TABLE  
		[ WITH <function_option> [ ,...n ] ]  
		[ AS ]  
		RETURN [ ( ] select_stmt [ ) ]  
	[ ; ]
